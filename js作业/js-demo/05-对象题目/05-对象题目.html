this题目:

第一题:

      var myObject = {
        foo : "bar",
        fun : function(){
          var self = this;
          console.log("outer fun: this.foo = " + this.foo);
          console.log("outer fun: self.foo = " + self.foo);
          
          (function(){
            console.log("inner fun : this.foo = " + this.foo);
            console.log("outer fun: self.foo = " + self.foo);
          })();
        }
      }
      myObject.fun();

      var hero = {

      console.log(hero.getName());
